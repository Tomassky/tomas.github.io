  • /www/class/DB.class.php
<?php

    class DB
    {
        public $host;
        public $username;
        public $password;
        public $database;

        //初始化数据库
        function __construct($host='',$username='',$password='',$database='')
        {
            $this->host = empty($host)? $GLOBALS['con']['host'] : $host;
            $this->username = empty($username)? $GLOBALS['con']['username'] : $username;
            $this->password = empty($password)? $GLOBALS['con']['password'] : $password;
            $this->database = empty($database)? $GLOBALS['con']['database'] : $database;
            $lin = mysqli_connect($this->host,$this->username,$this->password);
            $this->use_database($lin);
        }

        //使用数据库
        public function use_database($lin=''){
            mysqli_query($lin,"use php_mysql_demo");
            $list = mysqli_query($lin,"select * from admin");
            $list_data = mysqli_fetch_array($list,MYSQLI_ASSOC);  //数据的存储
            var_dump($list_data);  //打印数据库的第一行数据
            echo "</br>";
        }
    }

?>
