  1. json_encode()

    • 来将数组和对象，转换为json格式
    • ▲ json只接受utf-8编码的字符，所以json_encode()的参数必须是utf-8编码，否则会得到空字符或者null
    • 数组格式
      • 索引数组 -> 数组格式
      • 关联数组 -> 对象格式
      • 类 -> 对象格式
        • 除了公开变量，其他东西（）
<?php

   //索引数组格式转换为json格式

   $arr = array('one','two','three');
   echo json_encode($arr);  //输出为 ["one","two","three"]
   echo "</br>";


   //关联数组格式转换为json格式

   $arr1 = array('hi' => 'one', 'how' => 'are', 'are' => 'you' );
   echo json_encode($arr1);  //输出为 {"hi":"one","how":"are","are":"you"}
   echo "</br>";

   //类转换为json格式

   class Foo{
    const ERROR_CODE = '404';
    public $publice_ex = 'this is publice';
    private $private_ex = 'this is private';
    protected $protected_ex = 'this should be protected';
    public function getErrorCode(){
        return self::ERROR_CODE;
    }
   }

   $foo = new Foo();
   $foo_json = json_encode($foo);
   echo $foo_json;  //输出为 {"publice_ex":"this is publice"}

?>

  1. json_decode()函数

    • 将json文本转换为相应的PHP数据结构
    • ▲ 通常情况下，json_decode()总是返回一个PHP对象，而不是数组
    • ▲ json只能用来表示对象（object）和数组（array），如果对一个字符串或数值使用json_decode()，将会返回null
    • 常见注意事项：
      • json的分隔符（delimiter）只允许使用双引号，不能使用单引号
      • json名值对的”名”（冒号左边的部分），任何情况下都必须使用双引号
      • 最后一个值之后不能添加逗号
<?php

    $json = '{"foo":123,"fll":456}';
    $obj = json_decode($json);
    var_dump($obj);  //输出为 object(stdClass)#1 (2) { ["foo"]=> int(123) ["fll"]=> int(456) } 
    echo "</br>";

    //强制转化为关联数组

    $obj1 = json_decode($json,true);
    var_dump($obj1);  //输出为 array(2) { ["foo"]=> int(123) ["fll"]=> int(456) } 

?>
  1. php从数据库读取数据,并以json格式返回数据的方法
<?php

    $data = array();
    $json = '';
    class user{
        public $id;
        public $fname;
        public $lname;
        public $email;
        public $password;
    }

    $conn = mysqli_connect('127.0.0.1','root','root','php_mysql_demo');

    $sql = "select * from json";
    $result = $conn->query($sql);

    if($result){
        while($row = mysqli_fetch_array($result,MYSQLI_ASSOC))
        {
            $user = new user();
            $user->id = $row['id'];
            $user->fname = $row['fname'];
            $user->lname = $row['lname'];
            $user->email = $row['email'];
            $user->password = $row['password'];
            $data[] = $user;
        }
        $json = json_encode($data);
        echo $json;  //输出为 [{"id":"7","fname":"tomas","lname":"sky","email":"643008933@qq.com","password":"mzt"}]
    }else{
        echo "查询失败";
    }
?>
