#!/usr/bin/python
from scapy.all import*
import logging
logging.getLogger("scapy,runtime").setLevel(logging.ERROR)
import sys

if len(sys.argv)!=2:
    print "Usage - ./ttl_os.py [IP Address]"
    print "Example - ./ttl_os.py 192.168.100.1"
    print "Example will perform ttl analysize to attempt to determine whether the system is Windows or Linux/Unix"
    sys.exit()

ip=sys.argv[1]
ans=sr1(IP(dst="str(ip)")/ICMP(),timeout=1,verbose=0)
if ans==None:
    print "No response was returned"
elif int(ans[IP].ttl)<=64:
    print "Host is Linux/Unix"
else:
    print "Host is Windows"

防火墙几种情况：
Send         Response           Type
SYN          NO              
ACK          RST                Filtered

SYN          SYN+ACK/SYN+RST

SYN          NO             
ACK          NO                 Closed
